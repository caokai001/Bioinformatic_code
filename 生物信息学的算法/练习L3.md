def bp_number(A):
    '''
    return 返回的函数包括：
    num:breakpoint 
    up_strip:上升序列

    '''
    A.insert(0,0)
    A.append(max(A)+1)
    # [0, 8, 9, 3, 4, 7, 6, 5, 1, 2, 10, 11, 12]
    num=[]
    # num 储存第几个逗号是breakpoint
    for i in range(1,len(A)):
        if abs(A[i]-A[i-1]) != 1:
            num.append(i)
    down_strip=[]
    ##用断点来计算up
    for i in range(len(num)-1):
        if A[num[i]]-A[num[i+1]-1]>0:  ##等于0为升序
            down_strip.append([num[i],num[i+1]-1])
    
    return [num,down_strip]
